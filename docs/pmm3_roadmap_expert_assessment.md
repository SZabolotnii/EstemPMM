   ```r
   .pmm3_newton_raphson <- function(
     X, y, moments, 
     init = NULL,  # –Ø–∫—â–æ NULL, –≤–∏–∫–æ—Ä–∏—Å—Ç–∞—Ç–∏ OLS
     max_iter = 100,
     tol = 1e-6,

   ) {
     # –Ø–∫—â–æ init=NULL, –ø–æ—á–∞—Ç–∏ –∑ OLS –æ—Ü—ñ–Ω–æ–∫

       init <- solve(t(X) %*% X, t(X) %*% y)

     b <- init
     converged <- FALSE
     
     for (iter in 1:max_iter) {
       # –û–±—á–∏—Å–ª–∏—Ç–∏ –≥—Ä–∞–¥—ñ—î–Ω—Ç —Ç–∞ Hessian

       # –ü–µ—Ä–µ–≤—ñ—Ä–∫–∞ —Å–∏–Ω–≥—É–ª—è—Ä–Ω–æ—Å—Ç—ñ Hessian
       if (rcond(hess) < 1e-10) {

         # –î–æ–¥–∞—Ç–∏ regularization
         hess <- hess + diag(1e-8, nrow(hess))
       }
       
       # Newton step
       delta <- solve(hess, -grad)
       
       # Backtracking line search (–æ–ø—Ü—ñ–æ–Ω–∞–ª—å–Ω–æ)
       if (backtrack) {
         alpha <- 1.0
         # ... (—Å–∫–ª–∞–¥–Ω–∞ –ª–æ–≥—ñ–∫–∞ line search)
       }
       
       b_new <- b + alpha * delta
       
       # –ü–µ—Ä–µ–≤—ñ—Ä–∫–∞ –∑–±—ñ–∂–Ω–æ—Å—Ç—ñ
       if (norm(b_new - b, "2") < tol) {

         break
       }
       
       b <- b_new
     }
     
     if (!converged) {
       warning("Newton-Raphson –ù–ï –ó–ë–Ü–ì–°–Ø –∑–∞ ", max_iter, " —ñ—Ç–µ—Ä–∞—Ü—ñ–π!")
     }

   }
   ```

   ```r
     ols <- solve(t(X) %*% X, t(X) %*% y)
     results[[3]] <- .pmm3_newton_raphson(
       X, y, moments,
       init = ols + rnorm(length(ols), 0, sd(ols)*0.1)
     )
     
     # –í–∏–±—Ä–∞—Ç–∏ –Ω–∞–π–∫—Ä–∞—â–µ —Ä—ñ—à–µ–Ω–Ω—è (–Ω–∞–π–º–µ–Ω—à–µ –∑–Ω–∞—á–µ–Ω–Ω—è —Ñ—É–Ω–∫—Ü—ñ—ó)
     objective_values <- sapply(results, function(r) {
       .pmm3_objective(r$coefficients, X, y, moments)
     })
     
     best_idx <- which.min(objective_values)
     return(results[[best_idx]])
   }
   ```

**–û—Ü—ñ–Ω–∫–∞:** ‚úÖ **–¢–µ–æ—Ä–µ—Ç–∏—á–Ω–æ –∫–æ—Ä–µ–∫—Ç–Ω–æ**

**–ê–ª–µ –ø–æ—Ç—Ä–µ–±—É—î –µ–º–ø—ñ—Ä–∏—á–Ω–æ—ó –≤–∞–ª—ñ–¥–∞—Ü—ñ—ó:**

```r
# –†–ï–ö–û–ú–ï–ù–î–ê–¶–Ü–Ø:
# –°—Ç–≤–æ—Ä–∏—Ç–∏ –∫–æ–º–ø–ª–µ–∫—Å–Ω—É —Ñ—É–Ω–∫—Ü—ñ—é –¥–ª—è –æ—Ü—ñ–Ω–∫–∏ –µ—Ñ–µ–∫—Ç–∏–≤–Ω–æ—Å—Ç—ñ

.pmm3_efficiency_analysis <- function(
  data_list,  # –°–ø–∏—Å–æ–∫ –¥–∞—Ç–∞—Å–µ—Ç—ñ–≤ –∑ —Ä—ñ–∑–Ω–∏–º–∏ Œ≥‚ÇÑ, Œ≥‚ÇÜ
  n_replications = 1000
) {
  results <- data.frame(
    dataset = character(),
    gamma4 = numeric(),
    gamma6 = numeric(),
    g_theoretical = numeric(),
    g_empirical = numeric(),
    variance_ratio = numeric(),
    mse_ratio = numeric()
  )
  
  for (i in seq_along(data_list)) {
    dataset <- data_list[[i]]
    
    # Monte Carlo –ø–æ—Ä—ñ–≤–Ω—è–Ω–Ω—è
    mc_results <- replicate(n_replications, {
      # –ì–µ–Ω–µ—Ä—É—î–º–æ –¥–∞–Ω—ñ –∑ –≤—ñ–¥–æ–º–∏–º —Ä–æ–∑–ø–æ–¥—ñ–ª–æ–º
      sim_data <- .generate_symmetric_data(
        n = 100,
        gamma4 = dataset$gamma4,
        gamma6 = dataset$gamma6
      )
      
      # Fit OLS —Ç–∞ PMM3
      ols_fit <- lm(y ~ ., data = sim_data)
      pmm3_fit <- lm_pmm3(y ~ ., data = sim_data)
      
      # –ü–æ—Ä—ñ–≤–Ω—è—Ç–∏ –¥–∏—Å–ø–µ—Ä—Å—ñ—ó –æ—Ü—ñ–Ω–æ–∫
      c(
        var_ols = var(coef(ols_fit)),
        var_pmm3 = var(coef(pmm3_fit))
      )
    })
    
    # –û–±—á–∏—Å–ª–∏—Ç–∏ –µ–º–ø—ñ—Ä–∏—á–Ω–µ g
    g_empirical <- mean(mc_results["var_pmm3", ]) / 
                   mean(mc_results["var_ols", ])
    
    # –û–±—á–∏—Å–ª–∏—Ç–∏ —Ç–µ–æ—Ä–µ—Ç–∏—á–Ω–µ g
    g_theoretical <- 1 - dataset$gamma4^2 / 
                     (6 + 9*dataset$gamma4 + dataset$gamma6)
    
    # –î–æ–¥–∞—Ç–∏ –¥–æ results
    results <- rbind(results, data.frame(
      dataset = dataset$name,
      gamma4 = dataset$gamma4,
      gamma6 = dataset$gamma6,
      g_theoretical = g_theoretical,
      g_empirical = g_empirical,
      variance_ratio = g_empirical,
      mse_ratio = mean(mc_results["mse_pmm3", ]) / 
                  mean(mc_results["mse_ols", ])
    ))
  }
  
  return(results)
}
```

```r
# Test 1: Gaussian data (Œ≥‚ÇÑ=0, Œ≥‚ÇÜ=0)
# –û—á—ñ–∫—É–≤–∞–Ω–Ω—è: PMM3 –ø–æ–≤–∏–Ω–µ–Ω –¥–∞–≤–∞—Ç–∏ —Ä–µ–∑—É–ª—å—Ç–∞—Ç–∏ –±–ª–∏–∑—å–∫—ñ –¥–æ OLS
test_pmm3_gaussian <- function() {
  n <- 1000
  X <- cbind(1, rnorm(n))
  beta_true <- c(1, 2)
  epsilon <- rnorm(n)  # Gaussian!
  y <- X %*% beta_true + epsilon
  
  # Fit
  ols_fit <- lm(y ~ X[,2])
  pmm3_fit <- lm_pmm3(y ~ X[,2])
  
  # –¢–µ—Å—Ç: –∫–æ–µ—Ñ—ñ—Ü—ñ—î–Ω—Ç–∏ –ø–æ–≤–∏–Ω–Ω—ñ –±—É—Ç–∏ –º–∞–π–∂–µ –æ–¥–Ω–∞–∫–æ–≤—ñ
  expect_equal(coef(pmm3_fit), coef(ols_fit), tolerance = 0.05)
  
  # –¢–µ—Å—Ç: g_coefficient –ø–æ–≤–∏–Ω–µ–Ω –±—É—Ç–∏ ‚âà 1
  expect_equal(pmm3_fit@g_coefficient, 1.0, tolerance = 0.1)
}

test_pmm3_uniform <- function() {
  n <- 1000
  X <- cbind(1, rnorm(n))
  beta_true <- c(1, 2)
  epsilon <- runif(n, -sqrt(3), sqrt(3))  # Uniform with variance 1
  y <- X %*% beta_true + epsilon
  
  # Monte Carlo: –ø–æ—Ä—ñ–≤–Ω—è—Ç–∏ variance
  replicate(500, {
    # ... –≥–µ–Ω–µ—Ä–∞—Ü—ñ—è –¥–∞–Ω–∏—Ö ...
    # ... fit OLS —Ç–∞ PMM3 ...
    c(var_ols = var(coef(ols_fit)[2]), 
      var_pmm3 = var(coef(pmm3_fit)[2]))
  }) -> mc_results
  
  # –¢–µ—Å—Ç: PMM3 variance –ø–æ–≤–∏–Ω–Ω–∞ –±—É—Ç–∏ –º–µ–Ω—à–æ—é
  expect_true(mean(mc_results["var_pmm3",]) < mean(mc_results["var_ols",]))
  
  # –¢–µ—Å—Ç: –µ–º–ø—ñ—Ä–∏—á–Ω–µ g –±–ª–∏–∑—å–∫–µ –¥–æ —Ç–µ–æ—Ä–µ—Ç–∏—á–Ω–æ–≥–æ
  g_empirical <- mean(mc_results["var_pmm3",]) / mean(mc_results["var_ols",])
  g_theoretical <- 1 - (-1.2)^2 / (6 + 9*(-1.2) + (-1.0286))
  expect_equal(g_empirical, g_theoretical, tolerance = 0.15)
}
```

```r
# –ù–û–í–ò–ô –§–ê–ô–õ: R/pmm3_diagnostics.R

#' @title –î—ñ–∞–≥–Ω–æ—Å—Ç–∏–∫–∞ PMM3 fit
#' @description Comprehensive diagnostics –¥–ª—è –≤–∏—è–≤–ª–µ–Ω–Ω—è –ø—Ä–æ–±–ª–µ–º
diagnostic_pmm3 <- function(fit) {
  checks <- list()
  
  # Check 1: –°–∏–º–µ—Ç—Ä—ñ—è
  checks$symmetry <- list(
    gamma3 = fit@gamma3,
    is_symmetric = abs(fit@gamma3) < 0.5,
    warning = if (abs(fit@gamma3) > 0.5) {
      "–ü–û–ü–ï–†–ï–î–ñ–ï–ù–ù–Ø: –î–∞–Ω—ñ –Ω–µ—Å–∏–º–µ—Ç—Ä–∏—á–Ω—ñ! PMM3 –º–æ–∂–µ –±—É—Ç–∏ –Ω–µ–æ–ø—Ç–∏–º–∞–ª—å–Ω–∏–º."
    } else NULL
  )
  
  # Check 2: –°—Ç–∞–±—ñ–ª—å–Ω—ñ—Å—Ç—å Œî‚ÇÉ
  checks$stability <- list(
    delta3 = fit@delta3,
    is_stable = abs(fit@delta3) > 1e-6,
    warning = if (abs(fit@delta3) < 1e-6) {
      "–ü–û–ü–ï–†–ï–î–ñ–ï–ù–ù–Ø: Œî‚ÇÉ –±–ª–∏–∑—å–∫–æ –¥–æ –Ω—É–ª—è! –ß–∏—Å–µ–ª—å–Ω–∞ –Ω–µ—Å—Ç–∞–±—ñ–ª—å–Ω—ñ—Å—Ç—å."
    } else NULL
  )
  
  # Check 3: –ó–±—ñ–∂–Ω—ñ—Å—Ç—å
  checks$convergence <- list(
    converged = fit@convergence,
    iterations = fit@iterations,
    warning = if (!fit@convergence) {
      "–ü–û–ü–ï–†–ï–î–ñ–ï–ù–ù–Ø: Newton-Raphson –ù–ï –∑–±—ñ–≥—Å—è!"
    } else NULL
  )
  
  # Check 4: –ï—Ñ–µ–∫—Ç–∏–≤–Ω—ñ—Å—Ç—å
  checks$efficiency <- list(
    g_coefficient = fit@g_coefficient,
    expected_improvement = (1 - fit@g_coefficient) * 100,
    warning = if (fit@g_coefficient > 1.0) {
      "–ü–û–ü–ï–†–ï–î–ñ–ï–ù–ù–Ø: PMM3 –ì–Ü–†–®–ò–ô –∑–∞ OLS! –ü–µ—Ä–µ–≤—ñ—Ä—Ç–µ –¥–∞–Ω—ñ."
    } else if (fit@g_coefficient > 0.95) {
      "INFO: PMM3 –¥–∞—î –º—ñ–Ω—ñ–º–∞–ª—å–Ω–µ –ø–æ–∫—Ä–∞—â–µ–Ω–Ω—è. OLS –º–æ–∂–µ –±—É—Ç–∏ –¥–æ—Å—Ç–∞—Ç–Ω—ñ–º."
    } else NULL
  )
  
  # Print summary
  cat("\n=== PMM3 Diagnostic Summary ===\n")
  for (check_name in names(checks)) {
    check <- checks[[check_name]]
    if (!is.null(check$warning)) {
      cat("\n", check$warning, "\n")
    }
  }
  
  invisible(checks)
}
```
